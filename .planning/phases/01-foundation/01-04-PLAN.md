```typescript
import { NextRequest, NextResponse } from 'next/server';

export async function middleware(req: NextRequest) {
  // Better Auth stores the session token in a cookie named "better-auth.session_token" by default.
  // We check cookie presence here; layout.tsx does the authoritative getSession() call.
  const token = req.cookies.get('better-auth.session_token');
  if (!token) {
    const url = req.nextUrl.clone();
    url.pathname = '/login';
    url.searchParams.set('next', req.nextUrl.pathname);
    return NextResponse.redirect(url);
  }
  return NextResponse.next();
}

export const config = { matcher: ['/dashboard/:path*'] };
```

6. Create `src/app/actions/auth.ts`:
```typescript
'use server';
import { auth } from '@/lib/auth';
import { prisma } from '@/lib/db';
import { normalizeSlug, validateSlug } from '@/lib/slug';
import { redirect } from 'next/navigation';
import { headers } from 'next/headers';
import { z } from 'zod';

const SignUpSchema = z.object({
  email: z.string().email(),
  password: z.string().min(8).max(128),
  practiceName: z.string().min(2).max(100),
  slug: z.string().min(3).max(40),
});

export type SignUpResult =
  | { ok: true }
  | { ok: false; field: 'email' | 'password' | 'practiceName' | 'slug'; message: string };

export async function signUpClinicOwner(raw: z.input<typeof SignUpSchema>): Promise<SignUpResult> {
  const parsed = SignUpSchema.safeParse(raw);
  if (!parsed.success) {
    const issue = parsed.error.issues[0];
    return { ok: false, field: issue.path[0] as SignUpResult extends { field: infer F } ? F : never, message: issue.message };
  }
  const { email, password, practiceName, slug: rawSlug } = parsed.data;
  const slug = normalizeSlug(rawSlug);
  const slugCheck = validateSlug(slug);
  if (!slugCheck.ok) {
    const message = slugCheck.reason === 'reserved' ? 'That URL is taken. Try another.' : 'Use only lowercase letters, numbers, and hyphens.';
    return { ok: false, field: 'slug', message };
  }

  // Check slug uniqueness BEFORE creating the Better Auth user (atomicity concession —
  // there is a small race window; the @unique constraint is the final guard).
  const existingSlug = await prisma.clinic.findUnique({ where: { slug } });
  if (existingSlug) return { ok: false, field: 'slug', message: 'That URL is taken. Try another.' };

  // Create Better Auth user; returns session. If email exists, signUpEmail throws a known error.
  try {
    const result = await auth.api.signUpEmail({
      body: { email, password, name: practiceName },
      headers: await headers(),
      asResponse: false,
    });
    const userId = result.user.id;
    // Create Clinic atomically. If this fails, roll back the user.
    try {
      await prisma.clinic.create({
        data: { ownerUserId: userId, practiceName, slug, accentColor: 'sage' },
      });
    } catch (e) {
      // On clinic creation failure, delete the user to avoid an orphan.
      await prisma.user.delete({ where: { id: userId } }).catch(() => {});
      if ((e as { code?: string }).code === 'P2002') {
        return { ok: false, field: 'slug', message: 'That URL is taken. Try another.' };
      }
      throw e;
    }
  } catch (e) {
    const msg = (e as Error).message?.toLowerCase() ?? '';
    if (msg.includes('email') && (msg.includes('exist') || msg.includes('already') || msg.includes('taken'))) {
      return { ok: false, field: 'email', message: 'That email is already in use. Log in instead?' };
    }
    throw e;
  }

  redirect('/dashboard');
}
```

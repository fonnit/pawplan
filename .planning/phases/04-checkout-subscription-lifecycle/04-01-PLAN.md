```prisma
// ─── Phase 4 — Subscription member state (PAY-01..07, DASH-03, DASH-05) ─────
//
// A Member row is created from `checkout.session.completed` (see plan 04-03).
// Written ONLY by the webhook dispatcher and the cancelMember server action.
// UI + dashboard loaders must read via withClinic() so RLS enforces isolation.
//
// Invariants:
//   - `status` is the ENUM MemberStatus — NEVER a boolean (PAY-07).
//   - (clinicId, stripeSubscriptionId) is unique → webhook replay never
//     creates a duplicate member even if dispatch raced (PITFALLS.md #2).
//   - `canceledAt` is set optimistically when the owner clicks Cancel, and
//     confirmed by the customer.subscription.deleted webhook. `status` only
//     flips to `canceled` on the webhook; never on the UI click.
//   - `currentPeriodEnd` mirrors Stripe's current_period_end so the dashboard
//     can render cancellation countdown without a Stripe round-trip.
//
// Smart Retries are OFF per locked decision — payment_failed_at records the
// single failure that flipped the member to past_due. No retry schedule.
enum MemberStatus {
  active
  past_due
  canceled
}

model Member {
  id                   String       @id @default(uuid()) @db.Uuid
  clinicId             String       @db.Uuid
  clinic               Clinic       @relation(fields: [clinicId], references: [id], onDelete: Cascade)
  planTierId           String       @db.Uuid
  planTier             PlanTier     @relation(fields: [planTierId], references: [id], onDelete: Restrict)

  // Stripe linkage — mirrored from webhook events, never authored by UI.
  stripeCustomerId     String       // cus_…
  stripeSubscriptionId String       @unique // sub_… — unique across platform

  // Custom fields collected inside Stripe Checkout (BLDR-07).
  petName              String
  species              String       // "dog" | "cat" (Checkout custom-field dropdown)
  ownerEmail           String       // seeded from Checkout customer_email

  // Lifecycle.
  status               MemberStatus @default(active)
  currentPeriodEnd     DateTime?    // mirrored from subscription.current_period_end
  enrolledAt           DateTime     @default(now()) // first invoice.paid timestamp
  paymentFailedAt      DateTime?    // set by invoice.payment_failed (PAY-05, DASH-03)
  canceledAt           DateTime?    // set optimistically by owner click + confirmed by webhook (DASH-05)

  createdAt            DateTime     @default(now())
  updatedAt            DateTime     @updatedAt

  // Indexes optimized for the dashboard members page + webhook upsert.
  @@unique([clinicId, stripeSubscriptionId]) // webhook replay safety
  @@index([clinicId, status])                // filter past_due / active
  @@index([clinicId, paymentFailedAt])       // DASH-03 filterable failed flag
  @@index([planTierId])
}
```

- `onDelete: Restrict` on planTier so accidentally deleting a tier with active members is blocked (data integrity over convenience).

```typescript
// ─── Phase 4 — Subscription lifecycle (PAY-01..07, DASH-03, DASH-05) ────────

/**
 * Mirror of the Prisma MemberStatus enum as a TS string union.
 * Kept in lockstep with prisma/schema.prisma → enum MemberStatus.
 *
 * PAY-07: NEVER a boolean. Every codepath must match on this union.
 */
export type MemberStatus = 'active' | 'past_due' | 'canceled';

/**
 * Keys of Stripe Checkout `custom_fields` entries we send, and read back
 * from `checkout.session.completed`. Both sides of the wire must use these
 * exact string literals — centralising them here prevents typo drift.
 *
 * Stripe requires `key` to be snake_case, max 25 chars, [a-z0-9_].
 */
export const CHECKOUT_CUSTOM_FIELD_KEYS = {
  petName: 'pet_name',
  species: 'species',
} as const;
export type CheckoutCustomFieldKey =
  (typeof CHECKOUT_CUSTOM_FIELD_KEYS)[keyof typeof CHECKOUT_CUSTOM_FIELD_KEYS];

/**
 * Species dropdown options for Stripe Checkout's `custom_fields` of type
 * `dropdown`. Dogs + cats only — REQUIREMENTS.md §Out of Scope explicitly
 * excludes exotics, equine, livestock in v1.
 *
 * Stripe requires `value` to be snake_case lowercase; `label` is user-facing.
 */
export const SPECIES_OPTIONS = [
  { value: 'dog', label: 'Dog' },
  { value: 'cat', label: 'Cat' },
] as const;
export type Species = (typeof SPECIES_OPTIONS)[number]['value'];

/**
 * Metadata written on every Checkout Session's subscription_data.metadata.
 * These keys are read back verbatim by the webhook handler (plan 04-03) —
 * they are the join between Stripe's object graph and PawPlan's DB.
 *
 * clinicId + planTierId let checkout.session.completed locate the Member
 * insert target WITHOUT a Stripe API round-trip.
 */
export interface CheckoutSubscriptionMetadata {
  clinicId: string;
  planId: string;
  planTierId: string;
}

/**
 * Derive MemberStatus from a Stripe subscription. Single source of truth —
 * the webhook handler calls this when persisting, and the dashboard loader
 * calls this for on-the-fly refresh if needed.
 *
 * Smart Retries are OFF (locked product decision 2026-04-23), so we do NOT
 * special-case `incomplete` as a separate status — the first failure flips
 * straight to past_due.
 */
export function deriveMemberStatusFromSubscription(status: string): MemberStatus {
  switch (status) {
    case 'active':
    case 'trialing':
      return 'active';
    case 'past_due':
    case 'unpaid':
    case 'incomplete':
    case 'incomplete_expired':
      return 'past_due';
    case 'canceled':
      return 'canceled';
    default:
      // Unknown status = flag in dashboard. Fail-safe, not fail-open.
      return 'past_due';
  }
}
```

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-04-01-01 | Information Disclosure | Member table reads | mitigate | RLS strict-mode policy: unset GUC returns zero rows; every read wrapped in `withClinic()`. |
| T-04-01-02 | Tampering | status field flipping to a non-enum value | mitigate | Postgres enum `MemberStatus` rejects anything outside active/past_due/canceled at the DB layer; Prisma types mirror. |
| T-04-01-03 | Elevation of Privilege | `v_public_clinic_plans` accidentally extended to include members | accept | View owner is superuser; no migration in this plan touches the view; future plans must audit. Added as a lint item. |
| T-04-01-04 | Spoofing | Cross-tenant Member insert via a forged clinicId in code | mitigate | WITH CHECK clause on RLS policy: INSERT with clinicId ≠ current GUC is rejected at DB level even if the app layer is buggy. |
| T-04-01-05 | Repudiation | Staff claim they didn't flag a member past_due | accept | `paymentFailedAt` is webhook-written only; no human UI writes it. Audit log is the StripeEvent table (already persisted). |
